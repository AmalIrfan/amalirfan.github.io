Here is a Hello World program written in x64 GNU assembly using Linux syscalls 

.text 
  .global _start

  _start:
    mov $1, %rax
    mov $1, %rdi
    lea msg(%rip), %rsi
    mov $len, %rdx
    syscall

    mov $60, %rax
    mov $0, %rdi
    syscall
    
  msg:
    .ascii "Hello, World!\n"

  len = . - msg
